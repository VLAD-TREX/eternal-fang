generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL") // ← ДОБАВЬ ЭТУ СТРОКУ
}

model User {
  id        String   @id @default(cuid())
  username  String   @unique
  email     String   @unique
  games     Json?
  createdAt DateTime @default(now())
  @@map("users")
}
